activity_KoM_visit_hideout = {
	is_shown = {
		is_adult = yes
		is_ai = no
	}

	can_start_showing_failures_only = {
		NOT = { is_activity_type_on_cooldown = activity_KoM_visit_hideout }
		is_available_adult = yes
	}

	can_start = {
		trigger_if = { limit = { has_game_rule = KoM_visit_hideout_accessability_always } }
		trigger_else_if = {
			limit = { has_game_rule = KoM_visit_hideout_accessability_easier }
			OR = {
				learning >= 15
				KoM_is_a_mage_trigger = yes
				has_trait = scholar
			}
		}
		trigger_else = { 
			OR = {
				learning >= 25 
				KoM_is_a_mage_trigger = yes
				has_trait = scholar
			}
		}
	}

	is_valid = {
		NOT = {
			has_variable = activity_invalidated
		}
		scope:host = {
			is_playable_character = yes
			NOT = { is_incapable = yes }
		}
	}

	on_invalidated = {
		# Host becomes unlanded
		if = {
			limit = { scope:host = { is_incapable = yes } }
			scope:activity = { activity_type = { save_scope_as = activity_type } }
			scope:host = { trigger_event = activity_system.0330 }
			every_attending_character = {
				limit = { NOT = { this = scope:host } }
				trigger_event = activity_system.0331
			}
		}
	}

	on_host_death = {
		scope:activity = { set_variable = { name = activity_invalidated } }
	}

	wait_time_before_start = { days = 5 }

			trigger = {

			desc = activity_KoM_visit_hideout_visited_owned
		}
		triggered_desc = {
			trigger = {
				scope:host = { any_in_list = { variable = visited_hideouts this = root } }
				var:KoM_local_hideout ?= { var:hideout_type != flag:KoM_owned }
			}
			desc = activity_KoM_visit_hideout_visited_unknown
		}
		triggered_desc = {
			trigger = {
				NOT = { var:KoM_local_hideout ?= { var:hideout_owner ?= scope:host } }
				scope:host = { NOT = { any_in_list = { variable = visited_hideouts this = root } } }
				NOT = { scope:host = { any_in_list = { variable = visited_hideouts_known_nothing this = root } } }
			}
			desc = activity_KoM_visit_hideout_unvisited
		}
	}

		value = 25

			limit = {
				scope:host = { any_in_list = { variable = visited_hideouts_known_nothing this = root } }
			}
			value = 0
		}
